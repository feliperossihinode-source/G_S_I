firebase.auth().onAuthStateChanged(async user => { if (!user) { window.location.href = "index.html"; return; } const snapshot = await firebase.firestore() .collection("usuarios") .where("email", "==", user.email) .get(); const dados = snapshot.docs[0].data(); if (dados.tipo !== "admin") { alert("Acesso negado"); window.location.href = "dashboard.html"; } });
